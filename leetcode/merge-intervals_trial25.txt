class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:
        intervals.sort()
        mi = [intervals.pop(0)]

        while intervals:
            i = intervals.pop(0)
            m = mi[-1]
            
            if i[0] <= m[1]:
                mi[-1][1] = max(m[1], i[1])

            else:
                mi.append(i)

        return mi
