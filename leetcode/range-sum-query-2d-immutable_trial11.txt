class NumMatrix:

    def __init__(self, matrix: List[List[int]]):
        self.matrix = matrix
        # precalculate the rows & columns
        self.rows = [[0] + list(accumulate(row)) for row in matrix]
        # self.cols = []

    def sumRegion(self, row1: int, col1: int, row2: int, col2: int) -> int:
        res = 0
        for i in range(row1, row2 + 1, 1):
            res += self.rows[i][col2 + 1] - self.rows[i][col1]
        return res
