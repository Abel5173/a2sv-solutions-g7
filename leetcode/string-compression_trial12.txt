class Solution:
    def compress(self, chars: List[str]) -> int:
        w = 0
        i = 0
        n = len(chars)

        while i < n:
            char = chars[i]
            cnt = 0

            while i < n and char == chars[i]:
                i += 1
                cnt += 1
            
            chars[w] = char
            w += 1

            if cnt > 1:
                for c in str(cnt):
                    chars[w] = c
                    w += 1
            
        return w
