class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        v = set('aeiou')
        cnt = sum(1 for i in s[:k] if i in v)
        max_cnt = cnt
        for i in range(k, len(s)):
            if s[i] in v:
                cnt += 1
            if s[i-k] in v:
                cnt -= 1
            max_cnt = max(cnt, max_cnt)
            
        return max_cnt
