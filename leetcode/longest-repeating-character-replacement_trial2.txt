class Solution:
    def characterReplacement(self, s: str, k: int) -> int:
        i, j = 0, 0
        max_f = 0
        ls = 0
        f = {}
        while j < len(s):
            f[s[j]] = f.get(s[j], 0) + 1
            max_f = max(max_f, f[s[j]])
            ws = j - i + 1
            if ws - max_f > k:
                f[s[i]] -= 1
                i += 1

            ls = max(ls, j - i + 1)

            j += 1

        return ls
